 $mdr = new MDR_api();

 $mdr->addParam( "command", "dnssec_get_details"  );
 $mdr->addParam( "domein",  "mijndomeinreseller"  );
 $mdr->addParam( "tld",     "nl"                  );

 $mdr->DoTransaction();

 if( $mdr->Values[ "errcount" ] > 0 ) {
   print "Fout opgetreden:<BR>";

   for( $i=1;$i<=$mdr->Values[ "errcount" ];$i++ ) {
     print $mdr->Values[ "errnotxt".$i ] . " (code: " . $mdr->Values[ "errno".$i ] . ")<BR>";
   }

 } else {
  print "OK<BR>";
  print "<TABLE>";
  for($i=0;$i<$mdr->Values["recordcount"];$i++) {
    print "<TR>";
    print "<TD>" . $mdr->Values[ "record_id[$i]" ] . "</TD>";
    print "<TD>" . $mdr->Values[ "flag[$i]" ]      . "</TD>";
    print "<TD>" . $mdr->Values[ "algorithm[$i]" ]      . "</TD>";
    print "<TD>" . $mdr->Values[ "publickey[$i]" ]   . "</TD>";
    print "</TR>";
  }
  print "</TABLE>";
 }
